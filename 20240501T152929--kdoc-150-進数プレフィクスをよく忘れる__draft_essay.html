KDOC 150: 進数プレフィクスをよく忘れる

result=`ls | grep -oE 'kdoc-[0-9]+' | sed 's/kdoc-//' | sort -nr | head -n 1`
succ=$((result + 1))

echo $(cat <<EOF
(save-excursion
  (while (re-search-backward "KDOC 150" nil t)
    (replace-match "KDOC $succ")))
EOF
)

(save-excursion (while (re-search-backward "KDOC 150" nil t) (replace-match "KDOC 150")))
