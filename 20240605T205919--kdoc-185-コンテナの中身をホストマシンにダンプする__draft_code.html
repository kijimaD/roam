概要

コンテナのルートファイルシステムを、ホストマシンのファイルシステムにダンプして中身を観察する。 

mkdir -p bundle/rootfs
docker pull centos:8
docker run --rm --name tmp -d centos:8 sleep
docker export tmp | tar -xC bundle/rootfs
ls -al bundle/rootfs/
