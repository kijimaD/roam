* iframe用

このページにプレビュー機能を埋め込む用。非表示にしたい。

#+begin_export html
<iframe id="preview-frame" src=""></iframe>

<style>
iframe {
    width: 30%;
    height: 60%;
    border: 1px solid var(--bs-gray);
    position: absolute;
    z-index: 100;
    display: none;
}
</style>

<script>
const currentHost = location.host;
const currentPath = location.pathname;

let overLink = false;
let overIframe = false;
const iframe = document.getElementById('preview-frame');

// リンクのマウスイベント
document.querySelectorAll('a').forEach(link => {
    link.addEventListener('mouseenter', function(event) {
        const linkURL = new URL(this.href, location.href);
        // サイト内リンクだけiframeに表示
        if (linkURL.host === currentHost && linkURL.pathname != currentPath) {
            overLink = true;
            iframe.src = linkURL.pathname + linkURL.search + linkURL.hash;

            iframe.style.display = 'block';
            iframe.style.left = `${event.pageX+10}px`;
            iframe.style.top = `${event.pageY+2}px`;
        }
    });
    link.addEventListener("mouseleave", () => {
        overLink = false;
    });
});

// iframeのマウスイベント
iframe.addEventListener('mouseenter', () => {
    overIframe = true;
});
iframe.addEventListener('mouseleave', () => {
    overIframe = false;
    maybeHideIframe();
});

// どちらにもいなければiframeを隠す
function maybeHideIframe() {
    if (!overLink && !overIframe) {
        iframe.style.display = 'none';
    }
}
