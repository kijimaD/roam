概要

大文字–小文字のASCIIコードオフセットを覚えておかなくても、runeの値だけで計算できる。 

import "fmt"

func main() {
        fmt.Println(conv('a'))
        fmt.Println(conv('b'))
        fmt.Println(conv('z'))
}

func conv(c rune) string {
        return string(c+('A'-'a'))
}
