KDOC n: PDF本棚を作る

result=`ls | grep -oE 'kdoc-[0-9]+' | sed 's/kdoc-//' | sort -nr | head -n 1`
succ=$((result + 1))

echo $(cat <<EOF
(save-excursion
  (while (re-search-backward "KDOC n" nil t)
    (replace-match "KDOC $succ")))
EOF
)
