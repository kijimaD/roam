KDOC 148: 愛のあるリマスター

result=`ls | grep -oE 'kdoc-[0-9]+' | sed 's/kdoc-//' | sort -nr | head -n 1`
succ=$((result + 1))

echo $(cat <<EOF
(save-excursion
  (while (re-search-backward "KDOC 148" nil t)
    (replace-match "KDOC $succ")))
EOF
)
